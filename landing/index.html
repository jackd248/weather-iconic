Usage Examples

<!-- Include the stylesheet -->
<link rel="stylesheet" href="weather-iconic/style.css">

<!-- Use icons with CSS classes -->
<span class="weather-sun"></span>
<span class="weather-cloud-rain"></span>
import { WeatherSun, WeatherCloudRain } from 'weather-iconic/react'

function WeatherDisplay() {
  return (
    <div>
      <WeatherSun size={32} color="#80BBB2" />
      <WeatherCloudRain size={32} multiColor primaryColor="#80BBB2" />
    </div>
  )
}
<template>
  <div>
    <WeatherSun :size="32" color="#80BBB2" />
    <WeatherCloudRain :size="32" multi-color primary-color="#80BBB2" />
  </div>
</template>

<script>
import { WeatherSun, WeatherCloudRain } from 'weather-iconic/vue'
</script>

<!-- Method 1: Load sprite file -->
<script>
  // Load sprite into DOM
  fetch('weather-iconic/sprites.svg')
    .then(response => response.text())
    .then(svg => {
      const div = document.createElement('div')
      div.style.display = 'none'
      div.innerHTML = svg
      document.body.prepend(div)
    })
</script>

<!-- Method 2: Include sprite inline -->
<div style="display: none;">
  <svg><use href="weather-iconic/sprites.svg"></use></svg>
</div>

<!-- Use icons -->
<svg width="32" height="32" fill="#80BBB2">
  <use href="#weather-sun"></use>
</svg>
<svg width="32" height="32">
  <use href="#weather-cloud-rain"></use>
</svg>

<!-- React/Vue sprite utility -->
import { SpriteIcon, loadSprite } from 'weather-iconic/sprite'

// Load once in your app
loadSprite('/path/to/sprites.svg')

// Use anywhere
<SpriteIcon name="sun" size={32} color="#80BBB2" />
